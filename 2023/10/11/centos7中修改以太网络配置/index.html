centos7中修改以太网络配置
  1. 安装centos7后配置以太网络

安装centos7后配置以太网络

  1. 使用ip addr查看当前的网卡信息（例如：ens33）

  2. 使用 vi /etc/sysconfig/network-scripts/ifcfg-ens33修改为静态路由并添加ip、网关、dns

  3. TYPE="Ethernet" #网络类型（通常是Ethemet）
PROXY_METHOD="none"
BROWSER_ONLY="no"
BOOTPROTO="static" #IP的配置方法[none|static|bootp|dhcp]（引导时不使用协议|静态分配IP|BOOTP协议|DHCP协议）
DEFROUTE="yes"
IPV4_FAILURE_FATAL="no"
IPV6INIT="yes"
IPV6_AUTOCONF="yes"
IPV6_DEFROUTE="yes"
IPV6_FAILURE_FATAL="no"
IPV6_ADDR_GEN_MODE="stable-privacy"
NAME="ens33"
UUID="d15dce0b-c2f7-4dbd-9172-54b526a5c2c8"
DEVICE="ens33"   #接口名（设备,网卡）
ONBOOT="yes" #系统启动的时候网络接口是否有效（yes/no）

IPADDR=192.168.3.12 #本机地址
GATEWAY=192.168.3.1 #默认网关
NETMASK=255.255.255.0 #子网掩码
DNS1=114.114.114.114
DNS2=8.8.8.8

  4. 重启网络服务
    service network restart
